%  Compiles with standard pdfLaTeX out of the box (no external images
%  required except the optional title logo).
% =====================================================================

    \end{beamercolorbox}%
  \fi
}
\makeatother

% ====================
% Section divider slide (Aggie Blue background, gold rule)
% ====================
\AtBeginSection[]{%
  \begin{frame}[plain,noframenumbering]
    \begin{tikzpicture}[remember picture,overlay]
      \fill[aggieblue] (current page.south west) rectangle (current page.north east);
      \fill[aggiegold] ([yshift=-0.6cm]current page.center) rectangle ([yshift=-0.55cm,xshift=4cm]current page.center);
      \fill[aggiegold] ([yshift=-0.6cm]current page.center) rectangle ([yshift=-0.55cm,xshift=-4cm]current page.center);
      \node[anchor=center,text=white,font=\Raleway\bfseries\Huge]
        at ([yshift=0.4cm]current page.center) {\insertsectionhead};
    \end{tikzpicture}
  \end{frame}
}

% ====================
% Placeholder figure box — labeled gray panel that stands in for a real
% image so the template compiles with NO external files. Swap each call
% for \includegraphics[...]{your_figure.png} when you have the asset.
%   \placeholderfig{caption}            % default height
%   \placeholderfig[6cm]{caption}       % custom height
% ====================
\newcommand{\placeholderfig}[2][3.5cm]{%
  \begin{tikzpicture}
    \node[draw=aggieblue,line width=0.4pt,fill=boxgray,
          minimum width=0.92\linewidth,minimum height=#1,
          align=center,rounded corners,
          font=\Lato\itshape,text=coolgray]
      {[\,figure placeholder\,]\\#2};
  \end{tikzpicture}%
}

% Reusable gold "highlight" callout box (the deck's goal/takeaway box)
\newcommand{\highlightbox}[1]{%
  \setbeamercolor{boxblue}{bg=aggieblue,fg=white}%
  \begin{beamercolorbox}[wd=\textwidth,rounded=true,leftskip=0.7em,rightskip=0.7em,sep=0.6em]{boxblue}%
    \scriptsize\Lato\bfseries\color{white}#1%
  \end{beamercolorbox}%
}

% COE corner mark: small logo with a thin white border (matches the poster template's
% \logoleft treatment: \fboxsep=0pt, white \fbox around the College of Engineering logo).
\newcommand{\coemark}{{\setlength{\fboxsep}{0pt}\setlength{\fboxrule}{1pt}%
  \color{white}\fbox{\includegraphics[height=0.5cm]{coe_logo.jpg}}}}

  \begin{tikzpicture}[remember picture,overlay]
    \fill[aggieblue]
      ([yshift=1cm]current page.north west)
      rectangle
      ([yshift=-3.8cm]current page.north east);
    \fill[aggiegold]
      ([yshift=-3.8cm]current page.north west)
      rectangle
      ([yshift=-3.95cm]current page.north east);

          font=\Raleway\bfseries]
      at ([yshift=-1.6cm]current page.north)
      {\usebeamerfont{title}\inserttitle};
    \node[anchor=center,text=aggiegold,align=center,text width=14cm,
          font=\Lato]
      at ([yshift=-2.4cm]current page.north)
      {\usebeamerfont{subtitle}\insertsubtitle};
    \node[anchor=north,text=aggieblue,align=center,font=\Lato]
      at ([yshift=-5.4cm]current page.north)
      {\usebeamerfont{author}\insertauthor};
    \node[anchor=north,text=coolgray,align=center,font=\Lato]
      at ([yshift=-6.0cm]current page.north)
      {\usebeamerfont{institute}\insertinstitute};
    \node[anchor=north,text=coolgray,align=center,font=\Lato]
      at ([yshift=-6.9cm]current page.north)
      {\usebeamerfont{date}\insertdate};
  \end{tikzpicture}%
}
